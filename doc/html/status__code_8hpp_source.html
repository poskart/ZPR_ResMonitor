185 inline StatusCode status_code(const std::string &status_code_string) {
186  class StringToStatusCode: public std::unordered_map<std::string,
187  SimpleWeb::StatusCode> {
188  public:
189  StringToStatusCode() {
190  for (auto &status_code : status_code_strings())
191  emplace(status_code.second, status_code.first);
192  }
193  };
194  static StringToStatusCode string_to_status_code;
195  auto pos = string_to_status_code.find(status_code_string);
196  if (pos == string_to_status_code.end())
197  return StatusCode::unknown;
198  return pos->second;
199 }
